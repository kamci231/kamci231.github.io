Supabase SQL Setup Guide

연동 전, Supabase 대시보드 SQL Editor에서 아래 쿼리를 반드시 실행해 주세요.

-- 1. search_logs 테이블 생성
create table public.search_logs (
  id bigint generated by default as identity primary key,
  keyword text not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 2. popular_searches 뷰 생성 (Top 30 집계)
create or replace view public.popular_searches as
select keyword, count(*) as count
from public.search_logs
group by keyword
order by count desc
limit 30;

-- 3. 익명(Public) 쓰기 및 읽기 권한 허용 (RLS 설정 시)
alter table public.search_logs enable row level security;
create policy "Allow insert" on public.search_logs for insert with check (true);
create policy "Allow select" on public.search_logs for select using (true);
Firebase Realtime Database Rules Guide
